```csharp
namespace Orders.API.Tests.Shared.Fakes;

/// <summary>
/// Implementación Fake de IEventPublisher que captura eventos en memoria.
/// Permite verificar qué eventos se publicaron sin necesitar RabbitMQ real.
/// Se registra en OrdersApiFactory sustituyendo la implementación real.
/// </summary>
public class FakeEventPublisher : IEventPublisher
{
    private readonly List<object> _events = new();

    public IReadOnlyList<object> PublishedEvents => _events.AsReadOnly();

    public Task PublishAsync<T>(T @event, CancellationToken cancellationToken = default)
    {
        _events.Add(@event!);
        return Task.CompletedTask;
    }

    public IEnumerable<T> GetEvents<T>() => _events.OfType<T>();

    public void Clear() => _events.Clear();
}
```

Crear `tests/Orders.API.Tests/Shared/Fixtures/LocalDbFixture.cs`:

```csharp
using Microsoft.EntityFrameworkCore;

namespace Orders.API.Tests.Shared.Fixtures;

/// <summary>
/// Fixture de SQL Server LocalDB para tests de integración con BD real.
/// - Crea una BD con nombre único en LocalDB (evita conflictos entre test runs paralelos)
/// - Aplica todas las migrations para tener el schema completo
/// - Elimina la BD al finalizar para dejar LocalDB limpio
/// Usar con IClassFixture o [Collection] para compartir entre tests.
/// </summary>
public class LocalDbFixture : IAsyncLifetime
{
    private readonly string _testDbName = $"OrdersTestDb_{Guid.NewGuid():N}";

    public string ConnectionString =>
        $"Server=(localdb)\\MSSQLLocalDB;" +
        $"Database={_testDbName};" +
        $"Trusted_Connection=true;" +
        $"TrustServerCertificate=true;";

    public async Task InitializeAsync()
    {
        var options = new DbContextOptionsBuilder<OrderDbContext>()
            .UseSqlServer(ConnectionString)
            .Options;

        using var ctx = new OrderDbContext(options);
        // EnsureCreated crea el schema. En M6.1 se cambia por MigrateAsync.
        await ctx.Database.EnsureCreatedAsync();
    }

    public async Task DisposeAsync()
    {
        var options = new DbContextOptionsBuilder<OrderDbContext>()
            .UseSqlServer(ConnectionString)
            .Options;

        using var ctx = new OrderDbContext(options);
        await ctx.Database.EnsureDeletedAsync();
    }
}

/// <summary>
/// Collection para tests que comparten la misma instancia de LocalDbFixture.
/// Tests en la misma Collection NO se ejecutan en paralelo entre sí.
/// </summary>
[CollectionDefinition("SqlServerCollection")]
public class SqlServerCollection : ICollectionFixture<LocalDbFixture> { }
```

using Orders.API.Domain.Entities;
using Orders.API.Domain.ValueObjects;

namespace Orders.API.Tests.Shared.Builders;

/// <summary>
/// Builder para crear instancias de Order en tests.
/// Oculta la complejidad de construcción y hace los tests más legibles.
/// Si el constructor de Order cambia, solo hay que actualizar el Builder.
/// </summary>
public class OrderBuilder
{
    private Guid _customerId = Guid.NewGuid();
    private readonly List<OrderLine> _lines = new();

    public OrderBuilder WithCustomer(Guid customerId)
    {
        _customerId = customerId;
        return this;
    }

    public OrderBuilder WithLine(
        string productName = "Test Product",
        int quantity = 1,
        decimal price = 10.00m,
        string currency = "EUR")
    {
        _lines.Add(new OrderLine(
            productId: Guid.NewGuid(),
            productName: productName,
            quantity: quantity,
            unitPrice: new Money(price, currency)));
        return this;
    }

    public OrderBuilder WithManyLines(int count)
    {
        for (int i = 0; i < count; i++)
            WithLine($"Product {i + 1}");
        return this;
    }

    public Order Build()
    {
        // Si no se añadió ninguna línea, crear una por defecto
        if (_lines.Count == 0)
            WithLine();

        return Order.Create(_customerId, _lines);
    }
}
