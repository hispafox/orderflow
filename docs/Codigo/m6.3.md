# M6.3 — CQRS Avanzado y Read Models

## Branch

```bash
git checkout m6.2
git checkout -b m6.3
```

## Objetivo

Completar el patrón CQRS separando realmente el modelo de lectura del de escritura. El `ListOrdersHandler` actual usa EF Core sobre el aggregate completo — genera JOINs innecesarios y escala mal. En este módulo creamos `orders.OrderSummaries`, una tabla desnormalizada consultada con Dapper directamente, alimentada por proyectores que escuchan los eventos del Outbox. Resultado: queries **10x más rápidas** sin tocar el modelo de escritura.

```csharp
using Microsoft.EntityFrameworkCore;
using Microsoft.EntityFrameworkCore.Metadata.Builders;

namespace Orders.API.Infrastructure.Persistence.ReadModel;

/// <summary>
/// EF Core gestiona el ESQUEMA (migration). Los DATOS los gestiona Dapper (lecturas)
/// y los proyectores (escrituras). Esta configuración solo existe para la migration.
/// </summary>
public class OrderSummaryConfiguration : IEntityTypeConfiguration<OrderSummary>
{
    public void Configure(EntityTypeBuilder<OrderSummary> builder)
    {
        builder.ToTable("OrderSummaries", "orders");
        builder.HasKey(s => s.OrderId);
        builder.Property(s => s.OrderId).ValueGeneratedNever();

        builder.Property(s => s.CustomerEmail).HasMaxLength(256).IsRequired();
        builder.Property(s => s.Status).HasMaxLength(20).IsRequired();
        builder.Property(s => s.TotalAmount).HasPrecision(18, 2).IsRequired();
        builder.Property(s => s.Currency).HasMaxLength(3).IsRequired();
        builder.Property(s => s.FirstItemName).HasMaxLength(200);
        builder.Property(s => s.ShippingCity).HasMaxLength(100);
        builder.Property(s => s.CancellationReason).HasMaxLength(500);

        // ─── Índices para las queries más frecuentes ──────────────────────────
        builder.HasIndex(s => s.CustomerId)
               .HasDatabaseName("IX_OrderSummaries_CustomerId");
        builder.HasIndex(s => s.Status)
               .HasDatabaseName("IX_OrderSummaries_Status");
        builder.HasIndex(s => s.CreatedAt)
               .HasDatabaseName("IX_OrderSummaries_CreatedAt");
        // Índice compuesto — "mis pedidos en estado X" (query más frecuente)
        builder.HasIndex(s => new { s.CustomerId, s.Status })
               .HasDatabaseName("IX_OrderSummaries_CustomerId_Status");
    }
}
```

## Paso 7 — OrderReadRepository con Dapper (fichero completo)

Crear `src/Orders.API/Infrastructure/Persistence/ReadModel/OrderReadRepository.cs`:

```csharp
using System.Diagnostics;
using System.Diagnostics.Metrics;
using System.Text;
using Dapper;
using Microsoft.Data.SqlClient;
using Orders.API.API.DTOs.Responses;
using Orders.API.Application.Interfaces;
using Orders.API.Application.Queries;

namespace Orders.API.Infrastructure.Persistence.ReadModel;

/// <summary>
/// Implementación del Read Repository usando Dapper.
/// Sin EF Core tracking — SQL directo, sin overhead de materialización de aggregates.
/// Cada método crea su propia SqlConnection (el pool ADO.NET las reutiliza).
/// </summary>
public class OrderReadRepository : IOrderReadRepository
{
    private readonly string _connectionString;
    private readonly ILogger<OrderReadRepository> _logger;

    // Métricas OTel — visibles en el Aspire Dashboard
    private static readonly Meter   _meter         = new("Orders.API.ReadModel");
    private static readonly Histogram<double> _queryDuration =
        _meter.CreateHistogram<double>(
            "orders.read_model.query.duration.ms",
            unit:        "ms",
            description: "Duration of Read Model Dapper queries");

    public OrderReadRepository(
        IConfiguration                config,
        ILogger<OrderReadRepository>  logger)
    {
        _connectionString = config.GetConnectionString("OrdersDb")!;
        _logger           = logger;
    }

    // ─── GetByIdAsync ──────────────────────────────────────────────────────────

    public async Task<OrderDetailDto?> GetByIdAsync(
        Guid orderId, Guid requestingUserId, bool isAdmin,
        CancellationToken ct = default)
    {
        var sw  = Stopwatch.StartNew();
        try
        {
            await using var connection = new SqlConnection(_connectionString);

            // Multi-mapping: una fila por línea — Dapper ensambla el objeto anidado
            const string sql = """
                SELECT
                    o.Id, o.CustomerId, o.CustomerEmail,
                    o.Status, o.TotalAmount, o.TotalCurrency AS Currency,
                    o.CreatedAt, o.ConfirmedAt, o.CancelledAt,
                    o.CancellationReason,
                    o.Street, o.City, o.ZipCode, o.Country,
                    l.Id   AS LineId,
                    l.ProductId, l.ProductName,
                    l.Quantity,
                    l.UnitPriceAmount, l.UnitPriceCurrency
                FROM   orders.Orders o
                LEFT JOIN orders.OrderLines l ON l.OrderId = o.Id
                WHERE  o.Id = @OrderId
                  AND  o.IsDeleted = 0
                  AND  (@IsAdmin = 1 OR o.CustomerId = @UserId)
                """;

            OrderDetailDto? result = null;

            await connection.QueryAsync<OrderDetailDto, OrderLineDto, OrderDetailDto>(
                sql,
                (order, line) =>
                {
                    result ??= order;
                    if (line?.Id != Guid.Empty) result.Lines.Add(line);
                    return result;
                },
                splitOn: "LineId",
                param:   new { OrderId = orderId, UserId = requestingUserId,
                               IsAdmin = isAdmin ? 1 : 0 });

            return result;
        }
        finally
        {
            sw.Stop();
            _queryDuration.Record(sw.Elapsed.TotalMilliseconds,
                new TagList { { "query", "get_by_id" } });
        }
    }

    // ─── ListAsync ─────────────────────────────────────────────────────────────

    public async Task<PagedResult<OrderSummaryDto>> ListAsync(
        Guid?   customerId, string? status,
        int     page,       int    pageSize,
        CancellationToken   ct = default)
    {
        var sw = Stopwatch.StartNew();
        try
        {
            await using var connection = new SqlConnection(_connectionString);

            // Construir cláusula WHERE con parámetros tipados (no interpolados)
            var where      = new List<string>();
            var parameters = new DynamicParameters();

            if (customerId.HasValue)
            {
                where.Add("CustomerId = @CustomerId");
                parameters.Add("CustomerId", customerId.Value);
            }
            if (!string.IsNullOrWhiteSpace(status))
            {
                where.Add("Status = @Status");
                parameters.Add("Status", status);
            }

            var whereClause = where.Any()
                ? $"WHERE {string.Join(" AND ", where)}"
                : string.Empty;

            var skip = (page - 1) * pageSize;

            var countSql = $"SELECT COUNT(1) FROM orders.OrderSummaries {whereClause}";
            var dataSql  = $"""
                SELECT OrderId, CustomerId, CustomerEmail, Status,
                       TotalAmount, Currency, LinesCount, FirstItemName,
                       CreatedAt, ConfirmedAt, ShippingCity
                FROM   orders.OrderSummaries {whereClause}
                ORDER  BY CreatedAt DESC
                OFFSET {skip} ROWS FETCH NEXT {pageSize} ROWS ONLY
                """;

            var total = await connection.ExecuteScalarAsync<int>(countSql, parameters);
            var items = await connection.QueryAsync<OrderSummaryDto>(dataSql, parameters);

            return new PagedResult<OrderSummaryDto>(items.ToList(), total, page, pageSize);
        }
        finally
        {
            sw.Stop();
            _queryDuration.Record(sw.Elapsed.TotalMilliseconds,
                new TagList { { "query", "list_orders" } });
        }
    }

    // ─── SearchAsync ───────────────────────────────────────────────────────────

    public async Task<PagedResult<OrderSummaryDto>> SearchAsync(
        OrderSearchQuery  query,
        CancellationToken ct = default)
    {
        var sw = Stopwatch.StartNew();
        try
        {
            await using var connection = new SqlConnection(_connectionString);
            var sb         = new StringBuilder("SELECT * FROM orders.OrderSummaries WHERE 1=1");
            var parameters = new DynamicParameters();

            if (query.CustomerId.HasValue)
            {
                sb.Append(" AND CustomerId = @CustomerId");
                parameters.Add("CustomerId", query.CustomerId.Value);
            }
            if (!string.IsNullOrWhiteSpace(query.Status))
            {
                sb.Append(" AND Status = @Status");
                parameters.Add("Status", query.Status);
            }
            if (query.From.HasValue)
            {
                sb.Append(" AND CreatedAt >= @From");
                parameters.Add("From", query.From.Value);
            }
            if (query.To.HasValue)
            {
                sb.Append(" AND CreatedAt <= @To");
                parameters.Add("To", query.To.Value);
            }
            if (!string.IsNullOrWhiteSpace(query.ShippingCity))
            {
                sb.Append(" AND ShippingCity = @City");
                parameters.Add("City", query.ShippingCity);
            }
            if (query.MinTotal.HasValue)
            {
                sb.Append(" AND TotalAmount >= @MinTotal");
                parameters.Add("MinTotal", query.MinTotal.Value);
            }

            var countSql = sb.ToString().Replace("SELECT *", "SELECT COUNT(1)");
            var skip     = (query.Page - 1) * query.PageSize;
            sb.Append($" ORDER BY CreatedAt DESC OFFSET {skip} ROWS FETCH NEXT {query.PageSize} ROWS ONLY");

            var total = await connection.ExecuteScalarAsync<int>(countSql, parameters);
            var items = await connection.QueryAsync<OrderSummaryDto>(sb.ToString(), parameters);

            return new PagedResult<OrderSummaryDto>(items.ToList(), total, query.Page, query.PageSize);
        }
        finally
        {
            sw.Stop();
            _queryDuration.Record(sw.Elapsed.TotalMilliseconds,
                new TagList { { "query", "search_orders" } });
        }
    }

    // ─── GetStatsByCustomerAsync ───────────────────────────────────────────────

    public async Task<OrderStatsDto> GetStatsByCustomerAsync(
        Guid customerId, CancellationToken ct = default)
    {
        var sw = Stopwatch.StartNew();
        try
        {
            await using var connection = new SqlConnection(_connectionString);

            // Agregación pura en SQL — Dapper mapea el resultado único
            const string sql = """
                SELECT
                    COUNT(*)                                            AS TotalOrders,
                    SUM(CASE WHEN Status = 'Pending'   THEN 1 ELSE 0 END) AS PendingOrders,
                    SUM(CASE WHEN Status = 'Confirmed' THEN 1 ELSE 0 END) AS ConfirmedOrders,
                    ISNULL(SUM(TotalAmount), 0)                        AS TotalSpent,
                    ISNULL(MAX(TotalAmount), 0)                        AS LargestOrder,
                    MIN(CreatedAt)                                     AS FirstOrderDate,
                    MAX(CreatedAt)                                     AS LastOrderDate
                FROM orders.OrderSummaries
                WHERE CustomerId = @CustomerId
                """;

            return await connection.QuerySingleAsync<OrderStatsDto>(
                sql, new { CustomerId = customerId });
        }
        finally
        {
            sw.Stop();
            _queryDuration.Record(sw.Elapsed.TotalMilliseconds,
                new TagList { { "query", "stats_by_customer" } });
        }
    }
}
```

```csharp
using MassTransit;
using Microsoft.EntityFrameworkCore;
using OrderFlow.Contracts.Events.Orders;
using Orders.API.Infrastructure.Persistence;
using Orders.API.Infrastructure.Persistence.ReadModel;

namespace Orders.API.Infrastructure.Consumers.Projectors;

/// <summary>
/// Proyector que crea la fila en OrderSummaries cuando se crea un pedido.
/// Recibe el evento del Outbox (vía RabbitMQ) — ~5s después del POST /api/orders.
/// El Read Model es eventualmente consistente con el Write Model.
/// </summary>
public class OrderSummaryProjector : IConsumer<OrderCreated>
{
    private readonly OrderDbContext _dbContext;
    private readonly ILogger<OrderSummaryProjector> _logger;

    public OrderSummaryProjector(
        OrderDbContext dbContext,
        ILogger<OrderSummaryProjector> logger)
    {
        _dbContext = dbContext;
        _logger    = logger;
    }

    public async Task Consume(ConsumeContext<OrderCreated> context)
    {
        var msg = context.Message;

        // Idempotencia — mensajes duplicados (at-least-once delivery) son ignorados
        var exists = await _dbContext.OrderSummaries
            .AnyAsync(s => s.OrderId == msg.OrderId,
                      context.CancellationToken);

        if (exists)
        {
            _logger.LogWarning(
                "OrderSummary for {OrderId} already exists — skipping duplicate",
                msg.OrderId);
            return;
        }

        _dbContext.OrderSummaries.Add(new OrderSummary
        {
            OrderId       = msg.OrderId,
            CustomerId    = msg.CustomerId,
            CustomerEmail = msg.CustomerEmail,
            Status        = "Pending",
            TotalAmount   = msg.Total,
            Currency      = msg.Currency,
            LinesCount    = msg.Items.Count,
            FirstItemName = msg.Items.FirstOrDefault()?.ProductName,
            CreatedAt     = msg.CreatedAt,
            ShippingCity  = msg.ShippingAddress?.City
        });

        await _dbContext.SaveChangesAsync(context.CancellationToken);

        _logger.LogInformation(
            "[PROJECTOR] OrderSummary created for order {OrderId}", msg.OrderId);
    }
}
```

### OrderConfirmedProjector.cs

Crear `src/Orders.API/Infrastructure/Consumers/Projectors/OrderConfirmedProjector.cs`:

```csharp
using MassTransit;
using Microsoft.EntityFrameworkCore;
using OrderFlow.Contracts.Events.Orders;
using Orders.API.Infrastructure.Persistence;

namespace Orders.API.Infrastructure.Consumers.Projectors;

public class OrderConfirmedProjector : IConsumer<OrderConfirmed>
{
    private readonly OrderDbContext _dbContext;
    private readonly ILogger<OrderConfirmedProjector> _logger;

    public OrderConfirmedProjector(
        OrderDbContext dbContext,
        ILogger<OrderConfirmedProjector> logger)
    {
        _dbContext = dbContext;
        _logger    = logger;
    }

    public async Task Consume(ConsumeContext<OrderConfirmed> context)
    {
        var msg     = context.Message;
        var summary = await _dbContext.OrderSummaries
            .FirstOrDefaultAsync(s => s.OrderId == msg.OrderId,
                                 context.CancellationToken);

        if (summary is null)
        {
            // Race condition: el OrderSummaryProjector aún no procesó el OrderCreated.
            // MassTransit reintentará automáticamente (backoff exponencial).
            throw new InvalidOperationException(
                $"OrderSummary for {msg.OrderId} not found — will retry");
        }

        if (summary.Status == "Confirmed")
        {
            _logger.LogWarning(
                "OrderSummary {OrderId} already Confirmed — skipping", msg.OrderId);
            return;
        }

        summary.Status      = "Confirmed";
        summary.ConfirmedAt = DateTime.UtcNow;

        await _dbContext.SaveChangesAsync(context.CancellationToken);
        _logger.LogInformation(
            "[PROJECTOR] OrderSummary confirmed for {OrderId}", msg.OrderId);
    }
}
```

```csharp
        if (summary is null || summary.Status == "Cancelled") return;

        summary.Status             = "Cancelled";
        summary.CancelledAt        = DateTime.UtcNow;
        summary.CancellationReason = msg.Reason;

        await _dbContext.SaveChangesAsync(context.CancellationToken);
    }
}
```

## Paso 10 — RebuildOrderSummariesHandler

El Read Model puede reconstruirse en cualquier momento desde el Write Model. Crear `src/Orders.API/Infrastructure/Persistence/Rebuild/RebuildOrderSummariesHandler.cs`:

```csharp
using MediatR;
using Microsoft.EntityFrameworkCore;
using Orders.API.Infrastructure.Persistence;
using Orders.API.Infrastructure.Persistence.ReadModel;

namespace Orders.API.Infrastructure.Persistence.Rebuild;

/// <summary>
/// Reconstruye el Read Model desde el Write Model (fuente de verdad).
/// Útil cuando: se añade un campo nuevo al Read Model, el proyector tuvo un bug,
/// o el Read Model está desincronizado por cualquier razón.
/// </summary>
public record RebuildOrderSummariesCommand : IRequest;

public class RebuildOrderSummariesHandler : IRequestHandler<RebuildOrderSummariesCommand>
{
    private readonly OrderDbContext _dbContext;
    private readonly ILogger<RebuildOrderSummariesHandler> _logger;

    public RebuildOrderSummariesHandler(
        OrderDbContext dbContext,
        ILogger<RebuildOrderSummariesHandler> logger)
    {
        _dbContext = dbContext;
        _logger    = logger;
    }

    public async Task Handle(RebuildOrderSummariesCommand _, CancellationToken ct)
    {
        _logger.LogWarning("Rebuilding OrderSummaries Read Model from scratch...");

        // 1. Borrar el Read Model actual
        await _dbContext.Database.ExecuteSqlRawAsync(
            "DELETE FROM orders.OrderSummaries", ct);

        // 2. Reconstruir en lotes de 100 desde el Write Model
        var processed = 0;
        var batchSize = 100;

        while (true)
        {
            var orders = await _dbContext.Set<Domain.Entities.Order>()
                .Include("_lines")   // campo privado
                .OrderBy(o => o.CreatedAt)
                .Skip(processed)
                .Take(batchSize)
                .ToListAsync(ct);

            if (!orders.Any()) break;

            foreach (var order in orders)
            {
                _dbContext.OrderSummaries.Add(new OrderSummary
                {
                    OrderId        = order.Id,
                    CustomerId     = order.CustomerId,
                    CustomerEmail  = string.Empty,  // no disponible en aggregate
                    Status         = order.Status.Value,
                    TotalAmount    = order.Total.Amount,
                    Currency       = order.Total.Currency,
                    LinesCount     = order.Lines.Count,
                    FirstItemName  = order.Lines.FirstOrDefault()?.ProductName,
                    CreatedAt      = order.CreatedAt,
                    ConfirmedAt    = order.ConfirmedAt,
                    CancelledAt    = order.CancelledAt,
                    ShippingCity   = order.ShippingAddress?.City
                });
            }

            await _dbContext.SaveChangesAsync(ct);
            processed += orders.Count;
            _logger.LogInformation("Rebuilt {Count} order summaries", processed);
        }

        _logger.LogInformation(
            "OrderSummaries rebuild complete. Total: {Total}", processed);
    }
}
```

```csharp
using System.Net;
using System.Net.Http.Headers;
using System.Net.Http.Json;
using FluentAssertions;
using Orders.API.API.DTOs.Responses;
using Orders.API.Tests.Shared;
using Xunit;

namespace Orders.API.Tests.Integration.ReadModel;

/// <summary>
/// Tests E2E que verifican que el pedido creado aparece en el listado
/// después de que el proyector lo procesa (consistencia eventual).
/// En tests: el TestHarness de MassTransit procesa los eventos síncronamente
/// → no hay desfase de 5 segundos real.
/// </summary>
[Trait("Category", "Integration")]
public class EventualConsistencyTests : IClassFixture<OrdersApiFactory>
{
    private readonly HttpClient _client;

    private static readonly object ValidOrderRequest = new
    {
        customerEmail   = "john@techshop.es",
        items           = new[] { new { productId = Guid.NewGuid(), quantity = 1 } },
        shippingAddress = new
        {
            street = "Gran Vía 28", city = "Madrid", zipCode = "28013", country = "ES"
        }
    };

    public EventualConsistencyTests(OrdersApiFactory factory)
    {
        _client = factory.CreateClient();
        _client.DefaultRequestHeaders.Authorization =
            new AuthenticationHeaderValue("Bearer", JwtTestHelper.CustomerToken());
    }

    [Fact]
    public async Task CreateOrder_EventuallyAppearsInListing()
    {
        // 1. Crear el pedido (write side)
        var createResp = await _client.PostAsJsonAsync("/api/orders", ValidOrderRequest);
        createResp.StatusCode.Should().Be(HttpStatusCode.Created);
        var created = await createResp.Content.ReadFromJsonAsync<OrderDto>();

        // 2. Verificar que aparece en el listado (read side)
        // En tests con TestHarness el proyector corre síncronamente
        var appeared = await RetryUntilAsync(async () =>
        {
            var listResp = await _client.GetAsync("/api/orders/my?page=1&pageSize=50");
            if (!listResp.IsSuccessStatusCode) return false;
            var paged = await listResp.Content
                .ReadFromJsonAsync<PagedResult<OrderSummaryDto>>();
            return paged?.Items.Any(o => o.OrderId == created!.Id) == true;
        }, maxRetries: 10, delayMs: 500);

        appeared.Should().BeTrue(
            "el pedido debe aparecer en el listado tras la proyección");
    }

    private static async Task<bool> RetryUntilAsync(
        Func<Task<bool>> condition, int maxRetries, int delayMs)
    {
        for (int i = 0; i < maxRetries; i++)
        {
            if (await condition()) return true;
            await Task.Delay(delayMs);
        }
        return false;
    }
}
```

```
  ↓ INSERT orders.OrderSummaries
  (consistencia eventual: ~5-10s)
```

## Reglas fundamentales

```
1. Commands → IOrderRepository (EF Core, dominio rico)
2. Queries  → IOrderReadRepository (Dapper, solo lectura)
3. Commands NUNCA usan IOrderReadRepository (puede estar desactualizado)
4. Proyectores NUNCA usan IOrderRepository (solo actualizan el Read Model)
5. El Write Model es la fuente de verdad
6. El Read Model puede reconstruirse en cualquier momento
```
